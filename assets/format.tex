\newtcolorbox{tornpage}{%
	enhanced jigsaw, breakable, % allow page breaks
	frame hidden, % hide the default frame
	left=.7in,
	bottom=1.5em,
	top=1.5em,
	right=.5in,
	underlay={%
		\draw [
		fill=white, % fill paper
		draw=gray!90, % boundary colour
		decorate, % decoration
		decoration={random steps,segment length=2pt,amplitude=1pt},
		drop shadow, % shadow
		]
		% top line
		(frame.north west)--(frame.north east)--
		% right line
		(frame.north east)--(frame.south east)--
		% bottom line
		(frame.south east)--(frame.south west)--
		% left line
		(frame.south west)--(frame.north west);
	},
	overlay={%
		\begin{tcbclipinterior}
			%\shade[inner color=green!80!yellow,outer color=yellow!10!white]
			%(interior.north west) circle (2cm);
			\draw[help lines, yshift=-8pt, ystep=\baselineskip, xstep=\linewidth, gray!25 
			](frame.north east) grid (frame.south west);
			\draw[help lines, gray!80] ([xshift=.5in]interior.north west)--
			([xshift=.5in]interior.south west);
			\draw[help lines, gray!60] ([xshift=-.3in]interior.north east)--
			([xshift=-.3in]interior.south east);
	\end{tcbclipinterior}},
	% paragraph skips obeyed within tcolorbox
	parbox=false,
}
